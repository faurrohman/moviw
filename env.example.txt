# Supabase PostgreSQL Connection
# Dapatkan dari Supabase Dashboard: Settings → Database → Connection string

# Database Connection String (Transaction Mode - Recommended untuk Serverless/Vercel)
# Format: postgresql://postgres.[PROJECT-REF]:[PASSWORD]@aws-0-[region].pooler.supabase.com:6543/postgres
# Contoh: postgresql://postgres.nzcbrkfhydyrjcynljlu:Masuk12%2A@aws-1-ap-south-1.pooler.supabase.com:6543/postgres
DATABASE_URL=postgresql://postgres.[PROJECT-REF]:[PASSWORD]@aws-0-[region].pooler.supabase.com:6543/postgres

# Table Name (optional, default: movies)
TABLE_NAME=movies

# Catatan:
# - Ganti [PROJECT-REF] dengan project reference dari Supabase
# - Ganti [PASSWORD] dengan password database Anda
# - Ganti [region] dengan region Anda (misalnya: ap-south-1, ap-southeast-1)
# - Jika password mengandung karakter khusus (*, #, @), encode dengan URL encoding:
#   * menjadi %2A
#   * # menjadi %23
#   * @ menjadi %40
#   * Space menjadi %20
# - Password database berbeda dengan password account Supabase
# - Jangan commit file .env ke git!
# - Untuk Vercel deployment, set environment variables di Vercel Dashboard
